### Часть 1. Модели данных (C#)

Создадим классы для Команды и Матча.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

namespace FootballLeague
{
    // Модель команды
    public class Team
    {
        public int Id { get; set; }
        public string Name { get; set; }
        
        // Статистика
        public int Played { get; set; }
        public int Wins { get; set; }
        public int Draws { get; set; }
        public int Losses { get; set; }
        public int GoalsFor { get; set; }       // Забитые
        public int GoalsAgainst { get; set; }   // Пропущенные

        // Вычисляемые свойства
        public int GoalDifference => GoalsFor - GoalsAgainst; // Разница мячей
        public int Points => (Wins * 3) + (Draws * 1);        // Очки
    }

    // Модель матча
    public class Match
    {
        public int Id { get; set; }
        public int Round { get; set; } // Тур (1-38)
        public Team HomeTeam { get; set; }
        public Team AwayTeam { get; set; }
        public int HomeGoals { get; set; }
        public int AwayGoals { get; set; }
        public bool IsPlayed { get; set; }
    }
}
```

### Часть 2. Бизнес-логика (Расчеты и сортировка)

Класс, который управляет турниром: обрабатывает матчи, обновляет статистику и сортирует таблицу по правилам: *Очки -> Разница мячей -> Забитые мячи*.

```csharp
    public class LeagueManager
    {
        public List<Team> Teams { get; set; } = new List<Team>();
        public List<Match> Matches { get; set; } = new List<Match>();

        // Метод обработки результатов матча
        public void ProcessMatchResult(Match match)
        {
            if (!match.IsPlayed) return;

            match.HomeTeam.Played++;
            match.AwayTeam.Played++;

            match.HomeTeam.GoalsFor += match.HomeGoals;
            match.HomeTeam.GoalsAgainst += match.AwayGoals;

            match.AwayTeam.GoalsFor += match.AwayGoals;
            match.AwayTeam.GoalsAgainst += match.HomeGoals;

            if (match.HomeGoals > match.AwayGoals)
            {
                match.HomeTeam.Wins++;
                match.AwayTeam.Losses++;
            }
            else if (match.HomeGoals < match.AwayGoals)
            {
                match.AwayTeam.Wins++;
                match.HomeTeam.Losses++;
            }
            else
            {
                match.HomeTeam.Draws++;
                match.AwayTeam.Draws++;
            }
        }

        // Получение отсортированной турнирной таблицы
        public List<Team> GetStandings()
        {
            return Teams
                .OrderByDescending(t => t.Points)           // 1. По очкам
                .ThenByDescending(t => t.GoalDifference)    // 2. По разнице мячей
                .ThenByDescending(t => t.GoalsFor)          // 3. По забитым
                .ToList();
        }

        // Поиск личных встреч
        public List<Match> GetHeadToHead(Team team1, Team team2)
        {
            return Matches.Where(m => 
                (m.HomeTeam == team1 && m.AwayTeam == team2) || 
                (m.HomeTeam == team2 && m.AwayTeam == team1)
            ).ToList();
        }
    }
```

### Часть 3. Набор данных (Dataset)

Генератор 20 команд и случайного календаря на 380 матчей (38 туров).

```csharp
    public void GenerateMockData()
    {
        // 1. Создаем 20 команд
        string[] teamNames = { "Арсенал", "Астон Вилла", "Борнмут", "Брентфорд", "Брайтон", 
                               "Челси", "Кристал Пэлас", "Эвертон", "Фулхэм", "Ливерпуль", 
                               "Лутон Таун", "Манчестер Сити", "Манчестер Юнайтед", "Ньюкасл", 
                               "Ноттингем Форест", "Шеффилд", "Тоттенхэм", "Вест Хэм", "Вулверхэмптон", "Бернли" };

        for (int i = 0; i < 20; i++)
        {
            Teams.Add(new Team { Id = i + 1, Name = teamNames[i] });
        }

        // 2. Генерируем 380 матчей (каждый с каждым дома и на выезде)
        Random rnd = new Random();
        int matchId = 1;

        foreach (var homeTeam in Teams)
        {
            foreach (var awayTeam in Teams)
            {
                if (homeTeam != awayTeam)
                {
                    var match = new Match
                    {
                        Id = matchId++,
                        Round = rnd.Next(1, 39), // Упрощенное распределение по турам
                        HomeTeam = homeTeam,
                        AwayTeam = awayTeam,
                        HomeGoals = rnd.Next(0, 5), // Случайный счет
                        AwayGoals = rnd.Next(0, 4),
                        IsPlayed = true // Считаем, что матч сыгран
                    };
                    Matches.Add(match);
                    ProcessMatchResult(match); // Сразу обновляем статистику
                }
            }
        }
    }
```

### Как реализовать оставшиеся сложные функции:

1. **Графики изменения положения:**
   Потребуется установить NuGet пакет `LiveCharts.Wpf`. Для отрисовки нужно будет сохранять историю очков команды после каждого тура, а не только итоговый результат.
2. **Прогнозирование:**
   Создается "песочница" (глубокая копия списка команд). Пользователь вводит предполагаемые счета несыгранных матчей, вызывается `ProcessMatchResult` для копии, и на экран выводится виртуальная таблица.
3. **Поиск:**
   Легко реализуется через LINQ. Например, команды, забившие больше 50 голов: `Teams.Where(t => t.GoalsFor > 50).ToList()`.
